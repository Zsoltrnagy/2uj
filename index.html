    const span = Math.max(0, hi - lo);

    return { lo: lo + span * 0.35, hi: lo + span * 0.65 };

  function encodeForm(data){

    const parts = [];

    for(const k in data){

      if(!Object.hasOwn(data, k)) continue;

      let v = data[k];

      if(v === undefined || v === null) v = "";

      parts.push(encodeURIComponent(k) + "=" + encodeURIComponent(String(v)));

    }

    return parts.join("&");

  }

    }


    const subtype = $("service").value;


    /* CLEANING PRICE LOGIC */

    if(activeCategory === "cleaning"){


      const bedrooms = Number($("cleanBedrooms").value || 1);

      const scope = $("cleanScope").value;


      let baseLow = 0, baseHigh = 0;


      switch(scope){

        case "Standard":

          baseLow = 80 + bedrooms * 30;

          baseHigh = 120 + bedrooms * 40;

          break;

        case "Deep":

          baseLow = 150 + bedrooms * 45;

          baseHigh = 220 + bedrooms * 60;

          break;

        case "Move-in/out":

          baseLow = 200 + bedrooms * 55;

          baseHigh = 320 + bedrooms * 70;

          break;

      }


      // county adjustment

      baseLow *= COUNTY_MULT[county] || 1.0;

      baseHigh *= COUNTY_MULT[county] || 1.0;


      const c = applyConfidenceBand(baseLow, baseHigh);


      return {

        low: c.lo,

        high: c.hi,

        county,

        zip,

        cat: activeCategory,

        subtype,

        bedrooms,

        scope

      };

    }


    /* PET PRICE LOGIC */

    if(activeCategory === "pet"){


      const subtypeOpts = PET_OPTIONS[subtype] || [];

      const chosen = subtypeOpts.find(x => x.value === $("petOption").value) || subtypeOpts[0];


      const petCount = Math.min(4, Number($("petCount").value || 1));

      const extraMult = petCount === 1 ? 1 : petCount === 2 ? 1.35 : petCount === 3 ? 1.6 : 1.8;


      let low = chosen.low * extraMult;

      let high = chosen.high * extraMult;


      const c = applyConfidenceBand(low, high);


      return {

        low: c.lo,

        high: c.hi,

        county,

        zip,

        cat: activeCategory,

        subtype,

        petOption: chosen.label,

        petCount

      };

    }


    /* OTHER TRADES (size-based) */

    const rawSize = $("size").value.trim();

    const size = Number(rawSize);

    if(!size || size < 1){

      alert("Enter a realistic size.");

      return null;

    }


    let low = size * 1.5;

    let high = size * 3.5;


    // category multipliers

    const svcMult = { remodel: 1.25, plumbing: 1.15, hvac: 1.4 };

    low *= svcMult[activeCategory] || 1;

    high *= svcMult[activeCategory] || 1;


    // disposal special case

    if(activeCategory === "plumbing" && subtype === "Disposal Replacement"){

      low = 250 * size;

      high = 650 * size;

    }


    low *= COUNTY_MULT[county] || 1;

    high *= COUNTY_MULT[county] || 1;


    const c = applyConfidenceBand(low, high);


    return {

      low: c.lo,

      high: c.hi,

      county,

      zip,

      cat: activeCategory,

      subtype,

      size

    };

  }
