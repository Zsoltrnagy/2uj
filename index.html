      var base = {1:90, 2:120, 3:150, 4:190}[beds] || 120;

      var deepMult = type === "deep" ? 1.45 : 1.0;

      var moveMult = type === "move" ? 1.75 : 1.0;


      var mult = deepMult * moveMult;


      low = base * mult;

      high = base * mult * 1.35;

    }


    // PET CARE

    else if(category === "pet"){

      var opt = $("petOption").value;

      var count = Number($("petCount").value);

      if(count > 4) count = 4;


      var list = PET_OPTIONS[subtype];

      var found = list ? list.find(o => o.label === opt) : null;


      if(found){

        var mult = {1:1.0, 2:1.35, 3:1.6, 4:1.8}[count];

        low = found.low * mult;

        high = found.high * mult;

      } else {

        low = 30;

        high = 60;

      }

    }


    // DISPOSAL REPLACEMENT

    else if(subtype === "Disposal Replacement"){

      var qty = size < 1 ? 1 : size;

      low = qty * 250;

      high = qty * 650;

    }


    // REMODEL / PLUMBING / HVAC general formula

    else {

      if(size < 1){ alert("Please enter size."); return null; }


      low = size * 1.5;

      high = size * 3.6;


      // flooring material

      if(subtype === "Flooring"){

        var m = {

          "Tile":1.0, "Wood":1.12, "Natural stone":1.35, "Laminate":0.85, "Carpet":0.8

        }[$("flooringMaterial").value] || 1.0;

        low *= m;

        high *= m;

      }

    }


    // SERVICE LEVEL MULTIPLIERS

    var lvl = $("serviceLevel").value;

    var lvlMult = {Budget:0.92, Standard:1.0, Premium:1.12, Luxury:1.25}[lvl] || 1.0;

    low *= lvlMult; high *= lvlMult;


    // CONFIDENCE BAND

    var adjLow = low * 1.05;

    var adjHigh = high * 0.8;


    return {

      low: adjLow,

      high: adjHigh,

      category,

      subtype,

      zip,

      size,

      cleanBedrooms: $("cleanBedrooms").value,

      cleanScope: $("cleanScope").value,

      notes: $("notes").value,

      serviceLevel: lvl

    };

  }
